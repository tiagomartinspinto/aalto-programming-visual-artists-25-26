// File: sketch.js
let eyeSpace = 60;

function setup() {
  createCanvas(600, 600);
}

function draw() {
  background(30);

  for (let x = eyeSpace / 2; x < width; x += eyeSpace) {
    for (let y = eyeSpace / 2; y < height; y += eyeSpace) {

      let distance = dist(mouseX, mouseY, x, y);

      // Eye size depends on distance to mouse
      let eyeSize = map(distance, 0, 300, 60, 25);
      eyeSize = constrain(eyeSize, 25, 60);

      // Change eye color when mouse pressed
      if (mouseIsPressed) {
        fill(200, 40, 40); // red eyes
        eyeSize = 40;
      } else {
        fill(240);         // normal eyes
      }

      noStroke();
      ellipse(x, y, eyeSize, eyeSize * 0.6);

      let pupilOffset = eyeSize * 0.15;
      let pupilX, pupilY;

      if (mouseIsPressed) {  // pupils stare straight when pressed
        pupilX = x;
        pupilY = y;
      } else {
        // Pupils follow the mouse
        let distanceX = mouseX - x;
        let distanceY = mouseY - y;

        distanceX = constrain(distanceX, -pupilOffset, pupilOffset);
        distanceY = constrain(distanceY, -pupilOffset, pupilOffset);

        pupilX = x + distanceX;
        pupilY = y + distanceY;
      }

      fill(20);
      ellipse(pupilX, pupilY, eyeSize * 0.4, eyeSize * 0.4);
    }
  }
}
