// File: sketch.js
let gridSize = 10;
let tileSize;
let ellipseSize = 40;

let smallestEllipse = 10;

function setup() {
  createCanvas(600, 600);
  tileSize = width / gridSize;
  rectMode(CENTER);
  ellipseMode(CENTER);
}

function draw() {
  background(200);

  // Square grid - Loops to generate a grid
  for (let y = 0; y < gridSize; y++) {
    for (let x = 0; x < gridSize; x++) {

      // Calculate the center of the square
      let posX = x * tileSize + tileSize / 2;
      let posY = y * tileSize + tileSize / 2;

      // Distance to the mouse
      let d = dist(mouseX, mouseY, posX, posY);

      // Calculates size by distance (near mouse = big, far = small)
      let dynamicSize = map(d, 0, 200, tileSize, 0);

      // Constrain the size
      dynamicSize = constrain(dynamicSize, smallestEllipse, tileSize / 2);

      // Dynamic stroke weight
      let dynamicStroke = map(dynamicSize, 0, tileSize, 0.5, 5);

      // Grid
      stroke(0);
      strokeWeight(1);
      noFill();
      rect(posX, posY, tileSize, tileSize);

      // Dynamic ellipse
      stroke(0);
      strokeWeight(dynamicStroke);
      fill(mouseX, mouseY, 255);
      ellipse(posX, posY, dynamicSize, dynamicSize);
    }
  }
}
