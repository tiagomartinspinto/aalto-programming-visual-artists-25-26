// The main idea is that you have a little guy blowing a bubblegum. As the bubble covers the whole canvas it explodes, boom! 
// Oh, and also his eyes are following the cursor direction (no specific reason for that, just to practice)

// Press 'w' to blow the gum, once it explodes, press mouse. To reset the thing press 'r'

// Setting up custom font (I'll add font file to the submission, however it's not super important for the code to function, it will just look less cool)
let myFont;

let ellipseDiameter = 200;
let gumDiameter = 0;

// Setting variables, since I have a few different states depending on the action
let boom = false;
let gumOnFace = false;
let blowing = false;

// function preload() {
//   // Load font (make sure the font file is in your project folder)
//   myFont = loadFont("BadaBoom BB.ttf");
// }

function setup () {
  createCanvas(400, 400); 
  // textFont(myFont);
}

function draw() {
  background (197, 221, 252);
  
  if (boom) {
    drawBoom();
  } else {
    drawFace(200, 200, ellipseDiameter);
  }
}

function drawFace(x, y, diameter) {
  
  // Face
  fill(255, 220, 220);
  noStroke();
  ellipse(x, y, diameter, diameter);
  
  // Ears
  fill(255, 220, 220);
  noStroke();
  let earOffsetX = 100;
  let earOffsetY = 10;
  let earSize = 50;
  ellipse(x - earOffsetX, y - earOffsetY, earSize, earSize); 
  ellipse(x + earOffsetX, y - earOffsetY, earSize, earSize); 
  
  // Eyes
  fill(255);
  let eyeOffsetX = 20;
  let eyeOffsetY = 20;
  let eyeSize = 30;
  ellipse(x - eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize); 
  ellipse(x + eyeOffsetX, y - eyeOffsetY, eyeSize, eyeSize); 
  
  // Pupils
  fill(0);
  let pupilSize = 10;
  let leftEyeX = x - eyeOffsetX;
  let leftEyeY = y - eyeOffsetY;
  let rightEyeX = x + eyeOffsetX;
  let rightEyeY = y - eyeOffsetY;
  let maxMove = (eyeSize - pupilSize)/2; 
  
  // Left pupil
  let ld = dist(leftEyeX, leftEyeY, mouseX, mouseY);
  let pupilLX = leftEyeX + (mouseX - leftEyeX) * min(1, maxMove/ld);
  let pupilLY = leftEyeY + (mouseY - leftEyeY) * min(1, maxMove/ld);
  ellipse(pupilLX, pupilLY, pupilSize, pupilSize);
  
  // Right pupil
  let rd = dist(rightEyeX, rightEyeY, mouseX, mouseY);
  let pupilRX = rightEyeX + (mouseX - rightEyeX) * min(1, maxMove/rd);
  let pupilRY = rightEyeY + (mouseY - rightEyeY) * min(1, maxMove/rd);
  ellipse(pupilRX, pupilRY, pupilSize, pupilSize);
  
  // Mouth (changing shape when pressing 'w')
  let mouthSize = 12 - gumDiameter * 0.02;
  stroke(0);
  strokeWeight(2);
  
  if (blowing && !gumOnFace) {
    fill(255, 220, 220);
    ellipse(x, y + 10, mouthSize, mouthSize);
  } else {
    noFill();
    line(x - 10, y + 10, x + 10, y + 10);
  }
  
  // Bubblegum!
  if (!gumOnFace) {
    fill(255, 198, 237, 99);
    noStroke();
    ellipse(x, y + 10, gumDiameter, gumDiameter);
  }
  
  if (gumOnFace) {
    drawGumSplatter(x, y + 10);
  }
  
  if (gumDiameter > 390) {
    boom = true;
  }
}

// BOOM!
function drawBoom() { 
  fill(255, 0, 127);
  noStroke();
  rect(0, 0, width, height);
  
  fill(255);
  noStroke(); 
  ellipse(width / 2, height / 2, 350, 200);
  fill(0);
  textAlign(CENTER, CENTER);
  text("BOOM", width / 2, height / 2);
}
  
// Boom aftermath
function drawGumSplatter(x, y) {
  fill(255, 198, 237);
  noStroke();
  ellipse(x, y, 40, 25);
  ellipse(x - 15, y - 5, 15, 10);
  ellipse(x + 18, y + 3, 12, 9);
  ellipse(x - 5, y + 10, 10, 8);
  ellipse(x + 10, y - 8, 8, 6);
}

// Setting pressing key and mouse actions, as well as full reset 
function keyPressed() {  
  console.log("Bubble size: " + gumDiameter); //it also shows the size of the bubble, I used it to track the right diameter)
  
  if (key === 'w' && !boom) {
    gumDiameter += 10;
    blowing = true;
  }
  
  if (key === 'r') {
    reset();
  }
}

function mousePressed() { 
  if (boom) {
    boom = false;
    gumOnFace = true;
    gumDiameter = 0;
    blowing = false;
  }
}

function reset() {
  gumDiameter = 0;
  boom = false; 
  gumOnFace = false;
  blowing = false;
}
