class Shape {
  float x;
  float y;
  float size;

  // Constructor
  Shape(float startX, float startY, float startSize) {
    x = startX;
    y = startY;
    size = startSize;
  }

  // Method to draw the shape
  void display() {
    ellipse(x, y, size, size);
  }
}

Shape myShape;

void setup() {
  size(400, 400);

  // Create an object from the Shape class
  myShape = new Shape(200, 200, 50);
}

void draw() {
  background(255);

  // Ask the object to draw itself
  myShape.display();
}
