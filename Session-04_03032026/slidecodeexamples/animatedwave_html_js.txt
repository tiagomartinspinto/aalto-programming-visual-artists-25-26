  <script src="https://cdnjs.cloudflare.com/ajax/libs/p5.js/1.9.0/p5.min.js"></script>

<body>
  <script>
    let offset = 0;

    function setup() {
      createCanvas(1200, 800);
    }

    function draw() {
      background(0);
      stroke(255);
      noFill();

      for (let x = 0; x < width; x += 5) {
        let angle = x * 0.05 + offset;
        let y = height / 2 + sin(angle) * 50;
        ellipse(x, y, 10, 10);
      }

      offset += 0.05;
    }
  </script>
